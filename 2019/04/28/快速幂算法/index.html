快速幂算法

发表于 2019-04-28 | 分类于 算法 ， c++

#include< iostream>

#include< stdio.h>

#define ll long long
using namespace std;
int kpow(ll a,ll b);
int main()
{
kpow(2,32);
}
int kpow(ll a,ll b)//int kpow(lla,llb,int c)
{
ll ans=1,base=a,temp=b;
while(b)
{
if(b&1!=0)#任何大于0的数除以2以后，最终都会为1
ans=base;//ans=ansbase%c
base=base;//base=(basebase)%c
cout<<base<<” “<<b<<endl;
b>>=1;
}
printf(“%lld的%lld次方为：%lld\n”,a,temp,ans);
return ans;
}
快速幂本质上a^n=a^n/2*a^n/2,如果是取数取余数，见代码注释
